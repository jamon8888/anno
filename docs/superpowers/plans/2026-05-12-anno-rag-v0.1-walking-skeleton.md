**Goal:** Ship a single Rust binary `anno-rag` that ingests a French document folder, anonymizes PII, indexes in LanceDB, and answers semantic search queries — end-to-end on a real fixture.

**Architecture:** Single crate `crates/anno-rag` (no premature decomposition). Pipeline: kreuzberg extract → anno + cloakpipe detect → cloakpipe SqliteVault pseudonymize → candle embed (BGE-small multilingual) → LanceDB upsert. Search: query pseudonymize → embed → LanceDB native hybrid (FTS BM25 + vector via `RRFReranker`). CLI via `clap`, 2 subcommands. Async via tokio.

**Tech Stack:** Rust 1.81, edition 2024. `kreuzberg=4.9.7`, `lancedb=0.27.2`, `arrow=58`, `candle 0.10`, `tokio 1.51`, `clap 4`, `cloakpipe-core` (path-dep). Pinned versions per [spec §4](../specs/2026-05-12-anno-rag-v0.1-walking-skeleton.md).

```rust
//! Thin wrapper over cloakpipe SqliteVault.
//!
//! - 32-byte key derived from an OS keyring secret via Argon2id (production)
//!   or from a deterministic password (tests via `Vault::ephemeral_for_test`).
//! - Wrapped in `Arc<tokio::sync::Mutex<...>>` because cloakpipe's vault is `!Sync`.

use crate::error::{Error, Result};
use cloakpipe_core::vault_sqlite::SqliteVault;
use cloakpipe_core::{DetectedEntity, PseudoToken};
use std::path::Path;
use std::sync::Arc;
use tokio::sync::Mutex;

#[derive(Clone)]
pub struct Vault {
    inner: Arc<Mutex<SqliteVault>>,
}

impl Vault {
    /// Open or create the vault at `path` with the given 32-byte key.
    pub fn open(path: &Path, key: [u8; 32]) -> Result<Self> {
        let v = SqliteVault::open(path.to_str().ok_or_else(|| Error::Vault("invalid path".into()))?, key.to_vec())
            .map_err(|e| Error::Vault(e.to_string()))?;
        Ok(Self { inner: Arc::new(Mutex::new(v)) })
    }

    /// Open an ephemeral in-memory vault — for tests only.
    pub fn ephemeral_for_test() -> Self {
        let v = SqliteVault::ephemeral();
        Self { inner: Arc::new(Mutex::new(v)) }
    }

    /// Pseudonymize a list of entities in `text`, returning the rewritten string.
    /// Entities must be sorted by start offset, non-overlapping.
    pub async fn pseudonymize(&self, text: &str, entities: &[DetectedEntity]) -> Result<String> {
        let mut v = self.inner.lock().await;
        let result = cloakpipe_core::replacer::Replacer::pseudonymize(text, entities, &mut *v)
            .map_err(|e| Error::Vault(e.to_string()))?;
        Ok(result.text)
    }

    /// Reverse lookup — used by search to rehydrate a pseudonym in user-facing output.
    pub async fn lookup(&self, token: &str) -> Option<String> {
        let v = self.inner.lock().await;
        v.lookup(token).map(|s| s.to_string())
    }
}

/// Derive a 32-byte vault key from an OS-keyring secret using Argon2id.
/// Falls back to a deterministic key derived from a passphrase env var for dev/test.
pub fn derive_key_from_keyring() -> Result<[u8; 32]> {
    use argon2::{Argon2, PasswordHasher, password_hash::SaltString};
    use rand::rngs::OsRng;

    let secret = match std::env::var("ANNO_RAG_VAULT_PASSPHRASE") {
        Ok(s) => s,
        Err(_) => {
            // Production path: pull from OS keyring
            let entry = keyring::Entry::new("anno-rag", "vault-secret")
                .map_err(|e| Error::Vault(format!("keyring open: {e}")))?;
            match entry.get_password() {
                Ok(s) => s,
                Err(keyring::Error::NoEntry) => {
                    // First run — generate a random secret and store it
                    let s: String = (0..32).map(|_| rand::random::<u8>() as char as u8 as char).collect();
                    entry.set_password(&s).map_err(|e| Error::Vault(format!("keyring set: {e}")))?;
                    s
                }
                Err(e) => return Err(Error::Vault(format!("keyring get: {e}"))),
            }
        }
    };

    let salt = SaltString::generate(&mut OsRng);
    let argon = Argon2::default();
    let hash = argon.hash_password(secret.as_bytes(), &salt)
        .map_err(|e| Error::Vault(format!("argon2: {e}")))?;
    let hash_bytes = hash.hash.ok_or_else(|| Error::Vault("argon2 no hash".into()))?;
    let bytes = hash_bytes.as_bytes();

    let mut key = [0u8; 32];
    key.copy_from_slice(&bytes[..32]);
    Ok(key)
}
```

```rust
//! French PII detection: regex pack + anno gliner_pii backend.
//!
//! Combines:
//!   - Built-in FR regex (NIR, SIRET, SIREN, IBAN-FR, FR phone)
//!   - anno::StackedNER with gliner_pii backend (names, addresses, organizations, …)
//!   - cloakpipe's existing pattern detector (email, URL, IP, …)
//! Merges results, deduplicates overlapping spans, sorts by start.

use crate::error::{Error, Result};
use anno::{Model, StackedNER, Language};
use cloakpipe_core::{DetectedEntity, DetectionSource, EntityCategory};
use regex::Regex;
use std::sync::OnceLock;

/// FR regex pack — matches the 4 most common identifiers.
struct FrPatterns {
    nir: Regex,
    siret: Regex,
    iban_fr: Regex,
    phone_fr: Regex,
}

impl FrPatterns {
    fn get() -> &'static Self {
        static P: OnceLock<FrPatterns> = OnceLock::new();
        P.get_or_init(|| FrPatterns {
            // NIR (numéro de sécurité sociale) — 15 chars in known format
            nir: Regex::new(r"\b[12]\d{2}(0[1-9]|1[0-2])(2[AB]|\d{2})\d{3}\d{3}\d{2}\b").unwrap(),
            // SIRET — 14 digits (Luhn checked in code)
            siret: Regex::new(r"\b\d{14}\b").unwrap(),
            // IBAN-FR — FR + 25 chars total
            iban_fr: Regex::new(r"\bFR\d{2}\s?(?:\d{4}\s?){5}\d{3}\b").unwrap(),
            // FR phone — +33 or 0-prefix, 10 digits
            phone_fr: Regex::new(r"\b(?:\+33[\s\.\-]?|0)[1-9](?:[\s\.\-]?\d{2}){4}\b").unwrap(),
        })
    }
}

fn luhn(s: &str) -> bool {
    let digits: Vec<u32> = s.chars().filter_map(|c| c.to_digit(10)).collect();
    if digits.is_empty() { return false; }
    let total: u32 = digits.iter().rev().enumerate()
        .map(|(i, &d)| {
            if i % 2 == 1 {
                let doubled = d * 2;
                if doubled > 9 { doubled - 9 } else { doubled }
            } else { d }
        }).sum();
    total % 10 == 0
}

pub struct Detector {
    ner: StackedNER,
}

impl Detector {
    /// Build a new detector. Loads anno's gliner_pii backend (downloads weights on first use).
    pub fn new() -> Result<Self> {
        let ner = StackedNER::default();
        Ok(Self { ner })
    }

    /// Detect entities in `text`. Returns merged, deduplicated, sorted spans.
    pub fn detect(&self, text: &str) -> Result<Vec<DetectedEntity>> {
        let mut all = Vec::new();

        // 1. FR regex set
        let p = FrPatterns::get();
        for m in p.nir.find_iter(text) {
            all.push(DetectedEntity {
                original: m.as_str().to_string(),
                start: m.start(),
                end: m.end(),
                category: EntityCategory::Custom("NIR".into()),
                confidence: 1.0,
                source: DetectionSource::Pattern,
            });
        }
        for m in p.siret.find_iter(text) {
            if luhn(m.as_str()) {
                all.push(DetectedEntity {
                    original: m.as_str().to_string(),
                    start: m.start(),
                    end: m.end(),
                    category: EntityCategory::Custom("SIRET".into()),
                    confidence: 1.0,
                    source: DetectionSource::Pattern,
                });
            }
        }
        for m in p.iban_fr.find_iter(text) {
            all.push(DetectedEntity {
                original: m.as_str().to_string(),
                start: m.start(),
                end: m.end(),
                category: EntityCategory::Custom("IBAN_FR".into()),
                confidence: 1.0,
                source: DetectionSource::Pattern,
            });
        }
        for m in p.phone_fr.find_iter(text) {
            all.push(DetectedEntity {
                original: m.as_str().to_string(),
                start: m.start(),
                end: m.end(),
                category: EntityCategory::PhoneNumber,
                confidence: 0.95,
                source: DetectionSource::Pattern,
            });
        }

        // 2. anno NER — names, organizations, locations, …
        let anno_entities = self.ner.extract_entities(text, Some(Language::French))
            .map_err(|e| Error::Detect(e.to_string()))?;
        for e in anno_entities {
            all.push(DetectedEntity {
                original: e.text.clone(),
                start: e.start() as usize,
                end: e.end() as usize,
                category: map_anno_category(&e.entity_type),
                confidence: e.confidence as f64,
                source: DetectionSource::Ner,
            });
        }

        // 3. Sort + dedup overlaps (prefer Pattern > Ner on overlap)
        all.sort_by_key(|e| (e.start, std::cmp::Reverse(e.end - e.start)));
        let mut out: Vec<DetectedEntity> = Vec::new();
        for e in all {
            if let Some(last) = out.last() {
                if e.start < last.end {
                    continue; // overlaps with previous — drop
                }
            }
            out.push(e);
        }
        Ok(out)
    }
}

fn map_anno_category(t: &anno::EntityType) -> EntityCategory {
    match t {
        anno::EntityType::Person => EntityCategory::Person,
        anno::EntityType::Organization => EntityCategory::Organization,
        anno::EntityType::Location => EntityCategory::Location,
        _ => EntityCategory::Custom(format!("{:?}", t)),
    }
}
```

```rust
//! Embed text with intfloat/multilingual-e5-small (118M params, 384-dim, multilingual).
//!
//! Uses candle for in-process inference. Model loaded once at construction;
//! `embed_batch` is the hot path.

use crate::config::AnnoRagConfig;
use crate::error::{Error, Result};
use candle_core::{Device, DType, Tensor};
use candle_nn::VarBuilder;
use candle_transformers::models::bert::{BertModel, Config};
use std::path::PathBuf;
use std::sync::Arc;
use tokenizers::Tokenizer;

pub struct Embedder {
    model: BertModel,
    tokenizer: Tokenizer,
    device: Device,
    dim: usize,
}

impl Embedder {
    /// Load model. Downloads weights from HF Hub on first call, caches in `cfg.models_cache()`.
    pub async fn load(cfg: &AnnoRagConfig) -> Result<Self> {
        let device = Device::Cpu; // v0.1: CPU only (GPU opt-in in v0.2)
        let api = hf_hub::api::tokio::Api::new()
            .map_err(|e| Error::Embed(format!("hf-hub init: {e}")))?;
        let repo = api.model(cfg.embed_model.clone());

        let config_path = repo.get("config.json").await.map_err(|e| Error::Embed(e.to_string()))?;
        let tokenizer_path = repo.get("tokenizer.json").await.map_err(|e| Error::Embed(e.to_string()))?;
        let weights_path = match repo.get("model.safetensors").await {
            Ok(p) => p,
            Err(_) => repo.get("pytorch_model.bin").await.map_err(|e| Error::Embed(e.to_string()))?,
        };

        let config_json = std::fs::read_to_string(&config_path)?;
        let config: Config = serde_json::from_str(&config_json)
            .map_err(|e| Error::Embed(format!("config parse: {e}")))?;
        let tokenizer = Tokenizer::from_file(&tokenizer_path)
            .map_err(|e| Error::Embed(format!("tokenizer load: {e}")))?;

        let vb = unsafe {
            VarBuilder::from_mmaped_safetensors(&[weights_path], DType::F32, &device)
                .map_err(|e| Error::Embed(format!("var builder: {e}")))?
        };
        let model = BertModel::load(vb, &config)
            .map_err(|e| Error::Embed(format!("bert load: {e}")))?;

        Ok(Self { model, tokenizer, device, dim: cfg.embed_dim })
    }

    pub fn dim(&self) -> usize {
        self.dim
    }

    /// Embed a batch of strings. Returns a Vec<Vec<f32>> of length texts.len(), each of size `dim`.
    /// Texts are prefixed with "passage: " per e5 convention.
    pub fn embed_batch(&self, texts: &[String]) -> Result<Vec<Vec<f32>>> {
        let prefixed: Vec<String> = texts.iter().map(|t| format!("passage: {t}")).collect();
        let encs = self.tokenizer.encode_batch(prefixed, true)
            .map_err(|e| Error::Embed(format!("tokenize: {e}")))?;

        let max_len = encs.iter().map(|e| e.get_ids().len()).max().unwrap_or(0);
        let mut ids = Vec::with_capacity(texts.len() * max_len);
        let mut mask = Vec::with_capacity(texts.len() * max_len);
        for e in &encs {
            let pad = max_len - e.get_ids().len();
            ids.extend(e.get_ids().iter().map(|&x| x as i64));
            ids.extend(std::iter::repeat(0i64).take(pad));
            mask.extend(e.get_attention_mask().iter().map(|&x| x as i64));
            mask.extend(std::iter::repeat(0i64).take(pad));
        }

        let n = texts.len();
        let input_ids = Tensor::from_vec(ids, (n, max_len), &self.device)
            .map_err(|e| Error::Embed(format!("ids tensor: {e}")))?;
        let attn = Tensor::from_vec(mask, (n, max_len), &self.device)
            .map_err(|e| Error::Embed(format!("mask tensor: {e}")))?;
        let token_type = Tensor::zeros((n, max_len), DType::I64, &self.device)
            .map_err(|e| Error::Embed(format!("token_type tensor: {e}")))?;

        // Forward pass — returns last_hidden_state [n, seq, hidden]
        let out = self.model.forward(&input_ids, &token_type, Some(&attn))
            .map_err(|e| Error::Embed(format!("forward: {e}")))?;

        // Mean pooling weighted by attention mask
        let mask_f = attn.to_dtype(DType::F32).map_err(|e| Error::Embed(e.to_string()))?;
        let mask_f = mask_f.unsqueeze(2).map_err(|e| Error::Embed(e.to_string()))?;
        let masked = out.broadcast_mul(&mask_f).map_err(|e| Error::Embed(e.to_string()))?;
        let sum = masked.sum(1).map_err(|e| Error::Embed(e.to_string()))?;
        let counts = mask_f.sum(1).map_err(|e| Error::Embed(e.to_string()))?;
        let pooled = sum.broadcast_div(&counts).map_err(|e| Error::Embed(e.to_string()))?;

        // L2 normalize
        let norm = pooled.sqr().map_err(|e| Error::Embed(e.to_string()))?
            .sum(1).map_err(|e| Error::Embed(e.to_string()))?
            .sqrt().map_err(|e| Error::Embed(e.to_string()))?
            .unsqueeze(1).map_err(|e| Error::Embed(e.to_string()))?;
        let normed = pooled.broadcast_div(&norm).map_err(|e| Error::Embed(e.to_string()))?;

        let v: Vec<Vec<f32>> = normed.to_vec2().map_err(|e| Error::Embed(e.to_string()))?;
        Ok(v)
    }
}
```

```rust
//! LanceDB persistence: chunks table, upsert via merge_insert, native hybrid search.

use crate::config::AnnoRagConfig;
use crate::error::{Error, Result};
use arrow_array::{
    Array, FixedSizeBinaryArray, FixedSizeListArray, Float32Array,
    RecordBatch, RecordBatchIterator, StringArray, TimestampMicrosecondArray, UInt32Array,
};
use arrow_schema::{DataType, Field, Schema, TimeUnit};
use chrono::Utc;
use futures::TryStreamExt;
use lancedb::query::{ExecutableQuery, QueryBase};
use lancedb::{Connection, Table};
use sha2::{Digest, Sha256};
use std::sync::Arc;
use uuid::Uuid;

pub const TABLE_NAME: &str = "chunks";

pub fn chunks_schema(dim: usize) -> Arc<Schema> {
    Arc::new(Schema::new(vec![
        Field::new("chunk_id",     DataType::FixedSizeBinary(16), false),
        Field::new("doc_id",       DataType::FixedSizeBinary(16), false),
        Field::new("source_path",  DataType::Utf8,                false),
        Field::new("folder_path",  DataType::Utf8,                false),
        Field::new("chunk_idx",    DataType::UInt32,              false),
        Field::new("text_pseudo",  DataType::Utf8,                false),
        Field::new("page",         DataType::UInt32,              true),
        Field::new("char_start",   DataType::UInt32,              false),
        Field::new("char_end",     DataType::UInt32,              false),
        Field::new("text_hash",    DataType::FixedSizeBinary(32), false),
        Field::new("vector",
            DataType::FixedSizeList(Arc::new(Field::new("item", DataType::Float32, true)), dim as i32),
            false),
        Field::new("ingested_at",  DataType::Timestamp(TimeUnit::Microsecond, None), false),
    ]))
}

#[derive(Clone)]
pub struct Store {
    tbl: Table,
    dim: usize,
}

#[derive(Debug, Clone)]
pub struct ChunkRecord {
    pub doc_id: Uuid,
    pub source_path: String,
    pub folder_path: String,
    pub chunk_idx: u32,
    pub text_pseudo: String,
    pub page: Option<u32>,
    pub char_start: u32,
    pub char_end: u32,
    pub vector: Vec<f32>,
}

#[derive(Debug, Clone)]
pub struct SearchHit {
    pub doc_id: Uuid,
    pub chunk_id: Uuid,
    pub source_path: String,
    pub folder_path: String,
    pub chunk_idx: u32,
    pub text_pseudo: String,
    pub page: Option<u32>,
    pub char_start: u32,
    pub char_end: u32,
    pub score: f32,
}

impl Store {
    pub async fn open(cfg: &AnnoRagConfig) -> Result<Self> {
        let conn = lancedb::connect(cfg.index_path().to_str().ok_or_else(|| Error::Store("invalid index path".into()))?)
            .execute()
            .await
            .map_err(|e| Error::Store(format!("connect: {e}")))?;

        let names = conn.table_names().execute().await.map_err(|e| Error::Store(e.to_string()))?;
        let schema = chunks_schema(cfg.embed_dim);
        let tbl = if names.iter().any(|n| n == TABLE_NAME) {
            conn.open_table(TABLE_NAME).execute().await.map_err(|e| Error::Store(e.to_string()))?
        } else {
            let empty = RecordBatchIterator::new(std::iter::empty(), schema.clone());
            conn.create_table(TABLE_NAME, Box::new(empty)).execute().await.map_err(|e| Error::Store(e.to_string()))?
        };

        Ok(Self { tbl, dim: cfg.embed_dim })
    }

    pub async fn upsert(&self, records: Vec<ChunkRecord>) -> Result<()> {
        if records.is_empty() { return Ok(()); }
        let schema = chunks_schema(self.dim);

        let n = records.len();
        let now = Utc::now().timestamp_micros();

        let mut chunk_ids = Vec::with_capacity(n);
        let mut doc_ids = Vec::with_capacity(n);
        let mut source_paths = Vec::with_capacity(n);
        let mut folder_paths = Vec::with_capacity(n);
        let mut chunk_idxs = Vec::with_capacity(n);
        let mut texts = Vec::with_capacity(n);
        let mut pages = Vec::with_capacity(n);
        let mut char_starts = Vec::with_capacity(n);
        let mut char_ends = Vec::with_capacity(n);
        let mut hashes = Vec::with_capacity(n);
        let mut vectors = Vec::with_capacity(n * self.dim);

        for r in &records {
            // chunk_id = deterministic v5(doc_id, chunk_idx) so re-ingest is idempotent
            let key = format!("{}::{}", r.doc_id, r.chunk_idx);
            let cid = Uuid::new_v5(&Uuid::NAMESPACE_OID, key.as_bytes());

            let mut h = Sha256::new();
            h.update(r.text_pseudo.as_bytes());
            let digest = h.finalize();

            chunk_ids.push(cid.as_bytes().to_vec());
            doc_ids.push(r.doc_id.as_bytes().to_vec());
            source_paths.push(r.source_path.clone());
            folder_paths.push(r.folder_path.clone());
            chunk_idxs.push(r.chunk_idx);
            texts.push(r.text_pseudo.clone());
            pages.push(r.page);
            char_starts.push(r.char_start);
            char_ends.push(r.char_end);
            hashes.push(digest.to_vec());
            vectors.extend_from_slice(&r.vector);
        }

        let chunk_id_a  = FixedSizeBinaryArray::try_from_iter(chunk_ids.into_iter())?;
        let doc_id_a    = FixedSizeBinaryArray::try_from_iter(doc_ids.into_iter())?;
        let source_a    = StringArray::from(source_paths);
        let folder_a    = StringArray::from(folder_paths);
        let idx_a       = UInt32Array::from(chunk_idxs);
        let text_a      = StringArray::from(texts);
        let page_a      = UInt32Array::from(pages);
        let cs_a        = UInt32Array::from(char_starts);
        let ce_a        = UInt32Array::from(char_ends);
        let hash_a      = FixedSizeBinaryArray::try_from_iter(hashes.into_iter())?;
        let vec_field   = Arc::new(Field::new("item", DataType::Float32, true));
        let vec_values  = Float32Array::from(vectors);
        let vec_a       = FixedSizeListArray::try_new(vec_field, self.dim as i32, Arc::new(vec_values), None)?;
        let ts_a        = TimestampMicrosecondArray::from(vec![now; n]);

        let batch = RecordBatch::try_new(schema.clone(), vec![
            Arc::new(chunk_id_a),
            Arc::new(doc_id_a),
            Arc::new(source_a),
            Arc::new(folder_a),
            Arc::new(idx_a),
            Arc::new(text_a),
            Arc::new(page_a),
            Arc::new(cs_a),
            Arc::new(ce_a),
            Arc::new(hash_a),
            Arc::new(vec_a),
            Arc::new(ts_a),
        ])?;

        let iter = RecordBatchIterator::new(std::iter::once(Ok(batch)), schema);
        self.tbl
            .merge_insert(&["doc_id", "chunk_idx"])
            .when_matched_update_all(None)
            .when_not_matched_insert_all()
            .execute(Box::new(iter))
            .await
            .map_err(|e| Error::Store(format!("merge_insert: {e}")))?;

        Ok(())
    }

    pub async fn search(&self, query_text: &str, query_vec: &[f32], k: usize) -> Result<Vec<SearchHit>> {
        // Native hybrid: vector + FTS, fused via built-in RRFReranker
        let stream = self.tbl
            .query()
            .nearest_to(query_vec.to_vec())
            .map_err(|e| Error::Store(format!("nearest_to: {e}")))?
            .limit(k * 5)
            .execute()
            .await
            .map_err(|e| Error::Store(format!("query exec: {e}")))?;

        let batches: Vec<RecordBatch> = stream.try_collect().await.map_err(|e| Error::Store(e.to_string()))?;
        let mut hits = Vec::new();
        for batch in batches {
            for i in 0..batch.num_rows() {
                hits.push(batch_to_hit(&batch, i, 0.0)?);
            }
        }
        // v0.1: vector-only search. Hybrid FTS lands in Task 9 once the FTS index is built.
        let _ = query_text;
        hits.truncate(k);
        Ok(hits)
    }
}

fn batch_to_hit(b: &RecordBatch, i: usize, score: f32) -> Result<SearchHit> {
    let cid = b.column(0).as_any().downcast_ref::<FixedSizeBinaryArray>().unwrap();
    let did = b.column(1).as_any().downcast_ref::<FixedSizeBinaryArray>().unwrap();
    let src = b.column(2).as_any().downcast_ref::<StringArray>().unwrap();
    let fld = b.column(3).as_any().downcast_ref::<StringArray>().unwrap();
    let idx = b.column(4).as_any().downcast_ref::<UInt32Array>().unwrap();
    let txt = b.column(5).as_any().downcast_ref::<StringArray>().unwrap();
    let pg  = b.column(6).as_any().downcast_ref::<UInt32Array>().unwrap();
    let cs  = b.column(7).as_any().downcast_ref::<UInt32Array>().unwrap();
    let ce  = b.column(8).as_any().downcast_ref::<UInt32Array>().unwrap();

    let cid_bytes: [u8; 16] = cid.value(i).try_into().unwrap();
    let did_bytes: [u8; 16] = did.value(i).try_into().unwrap();

    Ok(SearchHit {
        chunk_id: Uuid::from_bytes(cid_bytes),
        doc_id: Uuid::from_bytes(did_bytes),
        source_path: src.value(i).to_string(),
        folder_path: fld.value(i).to_string(),
        chunk_idx: idx.value(i),
        text_pseudo: txt.value(i).to_string(),
        page: if pg.is_null(i) { None } else { Some(pg.value(i)) },
        char_start: cs.value(i),
        char_end: ce.value(i),
        score,
    })
}
```

```rust
//! Pipeline orchestration: ingest one doc end-to-end, search.

use crate::config::AnnoRagConfig;
use crate::detect::Detector;
use crate::embed::Embedder;
use crate::error::{Error, Result};
use crate::ingest;
use crate::store::{ChunkRecord, SearchHit, Store};
use crate::vault::Vault;
use std::path::Path;
use uuid::Uuid;

pub struct Pipeline {
    detector: Detector,
    vault: Vault,
    embedder: Embedder,
    store: Store,
    cfg: AnnoRagConfig,
}

impl Pipeline {
    pub async fn new(cfg: AnnoRagConfig, vault_key: [u8; 32]) -> Result<Self> {
        std::fs::create_dir_all(&cfg.data_dir).map_err(Error::from)?;
        let detector = Detector::new()?;
        let vault = Vault::open(&cfg.vault_path(), vault_key)?;
        let embedder = Embedder::load(&cfg).await?;
        let store = Store::open(&cfg).await?;
        Ok(Self { detector, vault, embedder, store, cfg })
    }

    /// Ingest a single file: extract → detect → pseudonymize → embed → store →
    /// write anonymized copy to `output_dir`.
    pub async fn ingest_one(&self, path: &Path, output_dir: &Path) -> Result<()> {
        let extracted = ingest::extract(path, &self.cfg).await?;
        let doc_id = Uuid::now_v7();
        let folder_path = path.parent()
            .map(|p| p.display().to_string())
            .unwrap_or_default();

        let mut records = Vec::with_capacity(extracted.chunks.len());
        let mut pseudo_chunks: Vec<String> = Vec::with_capacity(extracted.chunks.len());

        for chunk in &extracted.chunks {
            let entities = self.detector.detect(&chunk.text)?;
            let pseudo = self.vault.pseudonymize(&chunk.text, &entities).await?;
            pseudo_chunks.push(pseudo);
        }

        // Batch-embed all pseudonymized chunks at once for throughput
        let vectors = self.embedder.embed_batch(&pseudo_chunks)?;

        for (i, chunk) in extracted.chunks.iter().enumerate() {
            records.push(ChunkRecord {
                doc_id,
                source_path: extracted.source_path.clone(),
                folder_path: folder_path.clone(),
                chunk_idx: chunk.idx,
                text_pseudo: pseudo_chunks[i].clone(),
                page: chunk.page,
                char_start: chunk.char_start,
                char_end: chunk.char_end,
                vector: vectors[i].clone(),
            });
        }

        self.store.upsert(records).await?;

        // Write the anonymized copy
        std::fs::create_dir_all(output_dir).map_err(Error::from)?;
        let stem = path.file_stem().and_then(|s| s.to_str()).unwrap_or("doc");
        let out_path = output_dir.join(format!("{stem}.anon.md"));
        let full_anon = pseudo_chunks.join("\n\n");
        std::fs::write(&out_path, full_anon).map_err(Error::from)?;

        tracing::info!(path = %path.display(), chunks = extracted.chunks.len(), "ingested");
        Ok(())
    }

    /// Walk a folder (recursive optional), ingest every supported file.
    pub async fn ingest_folder(&self, folder: &Path, recursive: bool, output_dir: &Path) -> Result<usize> {
        let mut count = 0;
        let walker = if recursive {
            walkdir::WalkDir::new(folder).into_iter()
        } else {
            walkdir::WalkDir::new(folder).max_depth(1).into_iter()
        };
        for entry in walker.filter_map(|e| e.ok()) {
            if !entry.file_type().is_file() { continue; }
            let path = entry.path();
            let ext = path.extension().and_then(|e| e.to_str()).unwrap_or("").to_lowercase();
            if !matches!(ext.as_str(), "pdf" | "docx" | "pptx" | "xlsx" | "txt" | "md" | "html" | "eml" | "msg") {
                continue;
            }
            match self.ingest_one(path, output_dir).await {
                Ok(()) => count += 1,
                Err(e) => tracing::warn!(path = %path.display(), error = %e, "ingest skipped"),
            }
        }
        Ok(count)
    }

    /// Search: pseudonymize query → embed → store.search
    pub async fn search(&self, query: &str, top_k: usize) -> Result<Vec<SearchHit>> {
        let entities = self.detector.detect(query)?;
        let pseudo_q = self.vault.pseudonymize(query, &entities).await?;
        let qv = self.embedder.embed_batch(&[pseudo_q.clone()])?
            .into_iter().next()
            .ok_or_else(|| Error::Embed("empty embedder output".into()))?;
        self.store.search(&pseudo_q, &qv, top_k).await
    }
}
```

```rust
use anno_rag::{config::AnnoRagConfig, pipeline::Pipeline, vault::derive_key_from_keyring};
use clap::{Parser, Subcommand};
use std::path::PathBuf;

#[derive(Parser)]
#[command(name = "anno-rag", version, about = "Local GDPR-compliant document anonymizer + RAG")]
struct Cli {
    #[command(subcommand)]
    cmd: Cmd,
}

#[derive(Subcommand)]
enum Cmd {
    /// Ingest documents from a folder.
    Ingest {
        /// Folder to ingest.
        folder: PathBuf,
        /// Recurse into subfolders.
        #[arg(short, long, default_value_t = false)]
        recursive: bool,
        /// Where to write pseudonymized copies. Defaults to ~/.anno-rag/outputs.
        #[arg(short, long)]
        output: Option<PathBuf>,
    },
    /// Search the indexed corpus.
    Search {
        /// Query text.
        query: String,
        /// Number of results.
        #[arg(short = 'k', long, default_value_t = 10)]
        top_k: usize,
    },
}

#[tokio::main]
async fn main() -> anyhow::Result<()> {
    tracing_subscriber::fmt()
        .with_env_filter(tracing_subscriber::EnvFilter::from_default_env())
        .init();

    let cli = Cli::parse();
    let cfg = AnnoRagConfig::default();
    let key = derive_key_from_keyring()?;
    let pipeline = Pipeline::new(cfg.clone(), key).await?;

    match cli.cmd {
        Cmd::Ingest { folder, recursive, output } => {
            let out = output.unwrap_or_else(|| cfg.data_dir.join("outputs"));
            let n = pipeline.ingest_folder(&folder, recursive, &out).await?;
            println!("ingested {n} documents → {}", out.display());
        }
        Cmd::Search { query, top_k } => {
            let hits = pipeline.search(&query, top_k).await?;
            for (i, h) in hits.iter().enumerate() {
                println!("#{} score={:.3} page={:?}", i + 1, h.score, h.page);
                println!("    source: {}", h.source_path);
                println!("    text:   {}", truncate(&h.text_pseudo, 200));
                println!();
            }
        }
    }
    Ok(())
}

fn truncate(s: &str, n: usize) -> String {
    if s.chars().count() <= n { s.into() }
    else { format!("{}…", s.chars().take(n).collect::<String>()) }
}
```
