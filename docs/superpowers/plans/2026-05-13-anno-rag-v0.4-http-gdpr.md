# anno-rag v0.4 — HTTP + GDPR Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans. Steps use checkbox (`- [ ]`) syntax.

**Goal:** Ship HTTP API mirroring MCP + GDPR Art. 30 (audit), Art. 17 (forget), Art. 15 (find-subject) + TOML config + CLI flags + 2 MCP polish items.

**Architecture:** New modules `api.rs`, `audit/`, `subject.rs`, `forget.rs`. Reuses existing `Pipeline` from v0.2. HTTP via axum, audit via custom hash-chained JSONL + HMAC daily sign. No new crates.

**Tech Stack:** axum 0.8 (already workspace dep), tower-http (already there for cors+trace), printpdf 0.7 (new) for register PDF, sha2 (already there), hmac (already there).

**Prerequisites:** v0.2 merged to main (commit `09502ee2`). v0.3 lands or doesn't — v0.4 doesn't depend on v0.3 (it builds on v0.2 main directly).

- [ ] **Step 2: Implement**

  ```rust
  use crate::error::{Error, Result};
  use chrono::{DateTime, Utc};
  use serde::{Deserialize, Serialize};
  use sha2::{Digest, Sha256};
  use std::path::{Path, PathBuf};
  use tokio::fs::{File, OpenOptions};
  use tokio::io::{AsyncBufReadExt, AsyncWriteExt, BufReader};
  use uuid::Uuid;

  #[derive(Serialize, Deserialize, Debug, Clone)]
  pub struct AuditEvent {
      pub ts: DateTime<Utc>,
      pub op: String,
      pub actor: String,
      pub subject_tokens: Vec<String>,
      pub doc_id: Option<String>,
      pub details: serde_json::Value,
      pub trace_id: Uuid,
  }

  #[derive(Serialize, Deserialize)]
  struct AuditLine {
      #[serde(flatten)]
      event: AuditEvent,
      prev_hash: String,
  }

  pub struct AuditWriter {
      file: File,
      prev_hash: String,
      path: PathBuf,
  }

  #[derive(Debug)]
  pub struct ChainStatus {
      pub lines: u64,
      pub ok: bool,
      pub first_bad_line: Option<u64>,
  }

  impl AuditWriter {
      pub async fn open(audit_dir: &Path) -> Result<Self> {
          tokio::fs::create_dir_all(audit_dir).await?;
          let path = audit_dir.join("audit.jsonl");
          // Compute prev_hash = sha256 of last line, or zeros if empty
          let prev_hash = if path.exists() {
              Self::last_line_hash(&path).await?
          } else {
              "0".repeat(64)
          };
          let file = OpenOptions::new()
              .create(true).append(true).open(&path)
              .await?;
          Ok(Self { file, prev_hash, path })
      }

      pub async fn append(&mut self, event: &AuditEvent) -> Result<()> {
          let line = AuditLine {
              event: event.clone(),
              prev_hash: format!("sha256:{}", self.prev_hash),
          };
          let mut s = serde_json::to_string(&line)
              .map_err(|e| Error::Audit(format!("serialize: {e}")))?;
          s.push('\n');
          self.file.write_all(s.as_bytes()).await?;
          // Update prev_hash to sha256 of the bytes we just wrote
          let mut h = Sha256::new();
          h.update(s.as_bytes());
          self.prev_hash = format!("{:x}", h.finalize());
          Ok(())
      }

      pub async fn flush(&mut self) -> Result<()> {
          self.file.flush().await?;
          self.file.sync_all().await?;
          Ok(())
      }

      async fn last_line_hash(path: &Path) -> Result<String> {
          let content = tokio::fs::read_to_string(path).await?;
          let last = content.lines().last().unwrap_or("");
          if last.is_empty() {
              return Ok("0".repeat(64));
          }
          // Hash the line + the trailing newline that was written with it
          let mut to_hash = last.to_string();
          to_hash.push('\n');
          let mut h = Sha256::new();
          h.update(to_hash.as_bytes());
          Ok(format!("{:x}", h.finalize()))
      }

      pub async fn verify_chain(audit_dir: &Path) -> Result<ChainStatus> {
          let path = audit_dir.join("audit.jsonl");
          if !path.exists() {
              return Ok(ChainStatus { lines: 0, ok: true, first_bad_line: None });
          }
          let f = File::open(&path).await?;
          let mut r = BufReader::new(f);
          let mut expected_prev = "0".repeat(64);
          let mut line_no = 0u64;
          let mut buf = String::new();
          while r.read_line(&mut buf).await? > 0 {
              line_no += 1;
              let parsed: AuditLine = serde_json::from_str(buf.trim_end())
                  .map_err(|e| Error::AuditCorrupted {
                      line: line_no,
                      detail: format!("parse: {e}"),
                  })?;
              let want = format!("sha256:{}", expected_prev);
              if parsed.prev_hash != want {
                  return Ok(ChainStatus {
                      lines: line_no,
                      ok: false,
                      first_bad_line: Some(line_no),
                  });
              }
              // Compute hash for next iter
              let mut h = Sha256::new();
              h.update(buf.as_bytes());
              expected_prev = format!("{:x}", h.finalize());
              buf.clear();
          }
          Ok(ChainStatus { lines: line_no, ok: true, first_bad_line: None })
      }
  }
  ```

  ```rust
  async fn bearer_auth(
      State(token): State<Arc<String>>,
      headers: HeaderMap,
      req: Request,
      next: Next,
  ) -> std::result::Result<Response, StatusCode> {
      let Some(auth) = headers.get(AUTHORIZATION) else {
          return Err(StatusCode::UNAUTHORIZED);
      };
      let Ok(s) = auth.to_str() else { return Err(StatusCode::UNAUTHORIZED) };
      let Some(provided) = s.strip_prefix("Bearer ") else {
          return Err(StatusCode::UNAUTHORIZED);
      };
      // constant-time compare
      use subtle::ConstantTimeEq;
      if provided.as_bytes().ct_eq(token.as_bytes()).unwrap_u8() != 1 {
          return Err(StatusCode::UNAUTHORIZED);
      }
      Ok(next.run(req).await)
  }
  ```

  ```rust
          .map_err(|e| Error::Mcp(format!("start: {e}")))?;

      tokio::select! {
          r = service.waiting() => {
              r.map_err(|e| Error::Mcp(format!("loop: {e}")))?;
          }
          _ = tokio::signal::ctrl_c() => {
              tracing::info!("SIGINT received, flushing + exiting MCP cleanly");
              // service drops here → rmcp cleanup
          }
      }
      Ok(())
  }
  ```
